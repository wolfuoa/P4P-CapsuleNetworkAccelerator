cmake_minimum_required(VERSION 3.15)
project(relu_conv_2d)
find_package(common)
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_SOURCES xrelu_conv_2d.c)
collect (PROJECT_LIB_SOURCES xrelu_conv_2d_sinit.c)
collect (PROJECT_LIB_SOURCES xrelu_conv_2d_linux.c)
collect (PROJECT_LIB_SOURCES xrelu_conv_2d_g.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_HEADERS xrelu_conv_2d.h)
collect (PROJECT_LIB_HEADERS xrelu_conv_2d_hw.h)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
include_directories(${CMAKE_BINARY_DIR}/include)
add_library(relu_conv_2d STATIC ${_sources})
set_target_properties(relu_conv_2d PROPERTIES LINKER_LANGUAGE C)
